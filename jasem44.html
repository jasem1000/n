using MedicalLabManager.Data; using MedicalLabManager.Models; using MedicalLabManager.ViewModels.Base; using System; // مهم لإضافة هذا السطر using System.Collections.ObjectModel; using System.Linq; using System.Windows; namespace MedicalLabManager.ViewModels { public class PatientViewModel : ViewModelBase { private readonly ApplicationDbContext _context = new ApplicationDbContext(); private ObservableCollection _patients; public ObservableCollection Patients { get { return _patients; } set { _patients = value; OnPropertyChanged(); } } private string _medicalId; public string MedicalId { get => _medicalId; set { _medicalId = value; OnPropertyChanged(); } } private string _fullName; public string FullName { get => _fullName; set { _fullName = value; OnPropertyChanged(); } } public RelayCommand AddPatientCommand { get; } public PatientViewModel() { // تم وضع هذا السطر داخل دالة منفصلة لجعل الكود أنظف LoadData(); AddPatientCommand = new RelayCommand(AddPatient, CanAddPatient); } private void LoadData() { try // <-- بداية الكود المضاف { // سيحاول تحميل المرضى من قاعدة البيانات var patientList = _context.Patients.ToList(); Patients = new ObservableCollection(patientList); } catch (Exception ex) // <-- إذا حدث خطأ { // سيعرض رسالة بالخطأ بدلًا من إيقاف الواجهة MessageBox.Show($"حدث خطأ أثناء تحميل البيانات: {ex.Message}"); // سيقوم بإنشاء قائمة فارغة لتجنب المشاكل Patients = new ObservableCollection(); } // <-- نهاية الكود المضاف } private void AddPatient(object? parameter) { var newPatient = new Patient { MedicalId = this.MedicalId, FullName = this.FullName, DateOfBirth = System.DateTime.Now, Gender = "N/A" }; _context.Patients.Add(newPatient); _context.SaveChanges(); Patients.Add(newPatient); MessageBox.Show("تمت إضافة المريض بنجاح!"); } private bool CanAddPatient(object? parameter) { return !string.IsNullOrWhiteSpace(MedicalId) && !string.IsNullOrWhiteSpace(FullName); } } }
